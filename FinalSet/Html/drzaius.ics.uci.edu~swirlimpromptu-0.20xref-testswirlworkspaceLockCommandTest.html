58      {
59          String scope;

61          if(lockScope == LockMethod.SCOPE_EXCLUSIVE)
62          {
63              scope = "exclusive";
64          }
65          else
66          {
67              scope = "shared";
68          }
69  
70          return "<?xml version='1.0' encoding='utf-8'?>\n"
71                 + "<D:lockinfo xmlns:D='DAV:'>\n" + "\t<D:lockscope><D:" + scope
72                 + "/></D:lockscope>\n"
73                 + "\t<D:locktype><D:write/></D:locktype>\n" + "\t<D:owner>\""
74                 + owner + "\"</D:owner>\n" + "</D:lockinfo>";
75      }
